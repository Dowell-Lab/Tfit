676 int classifier::fit2(segment * data, vector<double> mu_seeds, int topology,
677  int elon_move ) {
678  //=========================================================================
679  //compute just a uniform model...no need for the EM
680  if (K == 0) {
681  ll = 0;
682  double l = (data->maxX - data->minX);
683  double pos = 0;
684  double neg = 0;
685  for (int i = 0; i < data->XN; i++) {
686  pos += data->X[1][i];
687  neg += data->X[2][i];
688  }
689  double pi = pos / (pos + neg);
690  for (int i = 0; i < data->XN; i++) {
691  if (pi > 0) {
692  ll += log(pi / l) * data->X[1][i];
693  }
694  if (pi < 1) {
695  ll += log((1 - pi) / l) * data->X[2][i];
696  }
697  }
698  components = new component[1];
699  return 1;
700  }
701  random_device rd;
702  mt19937 mt(rd());
703 
704  int add = noise_max > 0;
705  components = new component[K + add];
706  //===========================================================================
707  //initialize(1) components with user defined hyperparameters
708  for (int k = 0; k < K; k++) {
709  components[k].set_priors(ALPHA_0, BETA_0, ALPHA_1, BETA_1, ALPHA_2, ALPHA_3, data->N, K);
710  }
711 
712  //===========================================================================
713  //random seeding, initialize(2), center of pausing components
714  int i = 0;
715  double mu;
716  double mus[K];
717  for (int k = 0; k < K; k++) {
718  if (mu_seeds.size() > 0 ) {
719  i = sample_centers(mu_seeds , p);
720  mu = mu_seeds[i];
721  if (r_mu > 0) {
722  normal_distribution<double> dist_r_mu(mu, r_mu);
723  mu = dist_r_mu(mt);
724  }
725  } else {
726  normal_distribution<double> dist_MU((data->minX + data->maxX) / 2., r_mu);
727  mu = dist_MU(mt);
728  }
729 
730  mus[k] = mu;
731  if (mu_seeds.size() > 0 ) {
732  mu_seeds.erase (mu_seeds.begin() + i);
733  }
734  }
735  sort_vector(mus, K);
736  for (int k = 0; k < K; k++) { //random seeding, initialize(3) other parameters
737  components[k].initialize_bounds(mus[k],
738  data, K, data->SCALE , 0., topology, foot_print, data->maxX, data->maxX);
739 
740  }
741  sort_components(components, K);
742  for (int k = 0; k < K; k++) {
743  if (k > 0) {
744  components[k].reverse_neighbor = &components[k - 1];
745  } else {
746  components[k].reverse_neighbor = NULL;
747  }
748  if (k + 1 < K) {
749  components[k].forward_neighbor = &components[k + 1];
750  } else {
751  components[k].forward_neighbor = NULL;
752  }
753  }
754  if (add) {
755  components[K].initialize_bounds(0., data, 0., 0. , noise_max, pi, foot_print, data->minX, data->maxX);
756  }
757 
758  //===========================================================================
759  int t = 0; //EM loop ticker
760  double prevll = nINF; //previous iterations log likelihood
761  converged = false; //has the EM converged?
762  int u = 0; //elongation movement ticker
763  double norm_forward, norm_reverse, N; //helper variables
764 
765  while (t < max_iterations && not converged) {
766  //======================================================
767  //reset old sufficient statistics
768  for (int k = 0; k < K + add; k++) {
769  //components[k].print();
770  components[k].reset();
771  if (components[k].EXIT) {
772  converged = false, ll = nINF;
773  return 0;
774  }
775 
776  }
777 
778  //======================================================
779  //E-step, grab all the stats and responsibilities
780  ll = 0;
781  for (int i = 0; i < data->XN; i++) {
782  norm_forward = 0;
783  norm_reverse = 0;
784 
785  for (int k = 0; k < K + add; k++) { //computing the responsibility terms
786  if (data->X[1][i]) { //if there is actually data point here...
787  norm_forward += components[k].evaluate(data->X[0][i], 1);
788  }
789  if (data->X[2][i]) { //if there is actually data point here...
790  norm_reverse += components[k].evaluate(data->X[0][i], -1);
791  }
792  }
793  if (norm_forward > 0) {
794  ll += LOG(norm_forward) * data->X[1][i];
795  }
796  if (norm_reverse > 0) {
797  ll += LOG(norm_reverse) * data->X[2][i];
798  }
799 
800  //now we need to add the sufficient statistics, need to compute expectations
801  for (int k = 0; k < K + add; k++) {
802  if (norm_forward) {
803  components[k].add_stats(data->X[0][i], data->X[1][i], 1, norm_forward);
804  }
805  if (norm_reverse) {
806  components[k].add_stats(data->X[0][i], data->X[2][i], -1, norm_reverse);
807  }
808  }
809  }
810 
811  //======================================================
812  //M-step
813  N = 0; //get normalizing constant
814  for (int k = 0; k < K + add; k++) {
815  N += (components[k].get_all_repo());
816  }
817 
818  for (int k = 0; k < K + add; k++) {
819  components[k].update_parameters(N, K);
820  }
821 
822  if (abs(ll - prevll) < convergence_threshold) {
823  converged = true;
824  }
825  if (not isfinite(ll)) {
826  ll = nINF;
827  return 0;
828  }
829  //======================================================
830  //should we try to move the uniform component?
831  if (u > 200 ) {
832  sort_components(components, K);
833  //check_mu_positions(components, K);
834  if (elon_move) {
835  update_j_k(components, data, K, N);
836  update_l(components, data, K);
837  }
838  u = 0;
839  }
840 
841  u++;
842  t++;
843  prevll = ll;
844  }
845 
846 
847  return 1;
848 }
