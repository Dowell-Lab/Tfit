32  double start = X[t][0] - window, stop = X[t][1], n = 0;

34  while (t < N and stop + window > X[t][0] - window ) {
35  stop = X[t][1];

38  }
39  vector<double> row = {start , stop + window, S[0] / n, S[1] / n, S[2] / n };
